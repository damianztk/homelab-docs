#!/bin/bash
# ================================================================
# 🧩 Proxmox Backup Script – msmtp + HTML Report Edition
# ================================================================
# Autor: Damian (z pomocą GPT-5)
# Cel: Lekki, bezpieczny i czytelny system backupów VM/LXC z raportem HTML
# ================================================================

# === KONFIGURACJA UŻYTKOWNIKA ===
STORAGE="hdd-storage"                 # Nazwa storage w /etc/pve/storage.cfg
RETENTION=7                           # Ilość backupów do zachowania
LOG_DIR="/var/log/proxmox_backup"     # Ścieżka do logów
EMAIL_TO="admin@twojadomena.pl"       # Odbiorca raportu
FREE_LIMIT=90                         # Alert, jeśli użycie storage > 90%
HOSTNAME=$(hostname)

# === INICJALIZACJA ===
mkdir -p "$LOG_DIR"
LOGFILE="$LOG_DIR/backup_$(date '+%Y-%m-%d').log"
BACKUP_RESULTS=()

log() {
  echo "$(date '+%Y-%m-%d %H:%M:%S') $1" | tee -a "$LOGFILE"
}

# === WYSYŁKA MAILA (HTML) przez msmtp ===
send_mail_html() {
  SUBJECT="$1"
  BODY_HTML="$2"
  {
    echo "To: $EMAIL_TO"
    echo "Subject: $SUBJECT"
    echo "Content-Type: text/html; charset=UTF-8"
    echo ""
    echo "$BODY_HTML"
  } | msmtp -a default "$EMAIL_TO"
}

# === START BACKUPU ===
log "=== 🚀 START BACKUPU ==="

# === SPRAWDZENIE STORAGE ===
MOUNT_POINT=$(pvesm path "$STORAGE" 2>/dev/null)
if [ -z "$MOUNT_POINT" ]; then
  log "❌ Błąd: nie znaleziono storage '$STORAGE'"
  send_mail_html "❌ Proxmox Backup Error ($HOSTNAME)" "<h3 style='color:red'>Nie znaleziono storage '$STORAGE'</h3>"
  exit 1
fi

FREE_SPACE=$(df -h "$MOUNT_POINT" | awk 'NR==2 {print $4}')
USED_PERC=$(df "$MOUNT_POINT" | awk 'NR==2 {print $5}' | tr -d '%')
log "💾 Wolne miejsce na $STORAGE: $FREE_SPACE (użycie: ${USED_PERC}%)"

if [ "$USED_PERC" -ge "$FREE_LIMIT" ]; then
  ALERT_MSG="⚠️ Mniej niż $((100 - FREE_LIMIT))% wolnego miejsca!"
  log "$ALERT_MSG"
  BACKUP_RESULTS+=("ALERT|STORAGE|$STORAGE|$ALERT_MSG")
fi

# === FUNKCJA DO BACKUPU (wspólna dla LXC i VM) ===
backup_entity() {
  local TYPE=$1
  local ID=$2
  local NAME=$3
  local CMD="vzdump $ID --storage $STORAGE --compress zstd --mode snapshot --remove yes --prune-backups keep-last=$RETENTION --quiet 1"

  local START_TIME=$(date '+%H:%M:%S')
  log "🔹 Backup $TYPE $ID ($NAME)"

  if eval "$CMD"; then
    local END_TIME=$(date '+%H:%M:%S')
    BACKUP_RESULTS+=("OK|$TYPE|$ID|$NAME|$START_TIME → $END_TIME")
    log "✅ Backup $TYPE $ID zakończony pomyślnie"
  else
    local END_TIME=$(date '+%H:%M:%S')
    BACKUP_RESULTS+=("FAIL|$TYPE|$ID|$NAME|$START_TIME → $END_TIME")
    log "❌ Błąd backupu $TYPE $ID"
  fi
}

# === BACKUP LXC ===
for CTID in $(pct list | awk 'NR>1 {print $1}'); do
  NAME=$(pct config "$CTID" | grep '^hostname' | awk '{print $2}')
  backup_entity "LXC" "$CTID" "${NAME:-brak_nazwy}"
done

# === BACKUP VM ===
for VMID in $(qm list | awk 'NR>1 {print $1}'); do
  NAME=$(qm config "$VMID" | grep '^name' | awk '{print $2}')
  backup_entity "VM" "$VMID" "${NAME:-brak_nazwy}"
done

# === CZYSZCZENIE STARYCH LOGÓW ===
find "$LOG_DIR" -type f -name "backup_*.log" -mtime +30 -delete
log "🧹 Usunięto logi starsze niż 30 dni"

# === GENEROWANIE RAPORTU HTML ===
HTML_REPORT="<html><body style='font-family:Arial,sans-serif;'>
<h2 style='color:#2c3e50'>📦 Proxmox Backup Report – $(date '+%Y-%m-%d')</h2>
<p><b>Host:</b> $HOSTNAME<br>
<b>Storage:</b> $STORAGE ($MOUNT_POINT)<br>
<b>Wolne miejsce:</b> $FREE_SPACE (użycie ${USED_PERC}%)</p>
<table border='1' cellspacing='0' cellpadding='6' style='border-collapse:collapse;'>
<tr style='background:#2c3e50;color:white;'>
<th>Status</th><th>Typ</th><th>ID</th><th>Nazwa</th><th>Czas</th></tr>"

for entry in "${BACKUP_RESULTS[@]}"; do
  IFS='|' read -r STATUS TYPE ID NAME TIME <<< "$entry"
  case "$STATUS" in
    OK) COLOR="#27ae60"; ICON="✅" ;;
    FAIL) COLOR="#e74c3c"; ICON="❌" ;;
    ALERT) COLOR="#f1c40f"; ICON="⚠️" ;;
  esac
  HTML_REPORT+="<tr style='color:$COLOR'>
    <td>$ICON $STATUS</td><td>$TYPE</td><td>$ID</td><td>$NAME</td><td>$TIME</td></tr>"
done

HTML_REPORT+="</table>
<p style='color:#7f8c8d;font-size:12px;margin-top:20px;'>Automatyczny raport wygenerowany przez skrypt <i>proxmox_backup_msmtp.sh</i>.</p>
</body></html>"

# === WYSYŁKA RAPORTU ===
send_mail_html "🧩 Proxmox Backup Report – $(date '+%Y-%m-%d')" "$HTML_REPORT"

log "=== 🏁 KONIEC BACKUPU ==="
exit 0
